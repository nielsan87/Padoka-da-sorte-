package main import ( "fmt" "html/template" "math/rand" "net/http" "time" ) var symbols = []string{"🍞", "🥖", "🍰", "🧁", "🥐"} type SlotResult struct { Reel1 string Reel2 string Reel3 string Message string } var tpl = template.Must(template.New("slot").Parse(` Padoka da Sorte 🎰

Padoka da Sorte 🎰

{{.Reel1}} | {{.Reel2}} | {{.Reel3}}
{{if .Message}}
{{.Message}}
{{end}} `)) func spin() (string, string, string) { rand.Seed(time.Now().UnixNano()) return symbols[rand.Intn(len(symbols))], symbols[rand.Intn(len(symbols))], symbols[rand.Intn(len(symbols))] } func checkWin(r1, r2, r3 string) bool { return r1 == r2 && r2 == r3 } func handler(w http.ResponseWriter, r *http.Request) { data := SlotResult{ Reel1: "❓", Reel2: "❓", Reel3: "❓", Message: "", } if r.Method == http.MethodPost { r1, r2, r3 := spin() data.Reel1, data.Reel2, data.Reel3 = r1, r2, r3 if checkWin(r1, r2, r3) { data.Message = "🎉 Parabéns! Você ganhou na Padoka da Sorte! 🎉" } else { data.Message = "Tente de novo! 🍞🥖🍰🧁🥐" } } tpl.Execute(w, data) } func main() { http.HandleFunc("/", handler) fmt.Println("Servidor rodando em http://localhost:8080") http.ListenAndServe(":8080", nil) }
