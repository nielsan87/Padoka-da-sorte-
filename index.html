import random def jogo_repositor(): saldo = 1000 itens = ['Tela', 'Bateria', 'Chip', 'Cabo', 'Placa', 'Carregador', 'Botão'] print("=== Jogo de Aposta Repositor Celular & PC ===") print(f"Saldo inicial: R${saldo}") while saldo > 0: print(f"\nSaldo atual: R${saldo}") try: aposta = int(input("Quanto deseja apostar? Tente novamente.") continue except ValueError: print("Digite um número válido.") continue resultado = [random.choice(itens) for _ in range(3)] print("Itens sorteados:", resultado) if resultado[0] == resultado[1] == resultado[2]: ganho = aposta * 5 saldo += ganho print(f"Trinca!
